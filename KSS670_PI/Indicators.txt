#Internal indicator functions
<Function KSS670_PI_InternalInd>
	#General parameters
	Set Direction = %param1%
	Set CenterX = %param2%
	Set CenterZ = %param3%
	Set Offset = %param4%

	#Floor indicator texture prefix
	Set FLInd-Prefix = %param5%

	#Directional arrows (Lit up when the elevator is moving)
	Set ArrowBlankTex = %param6%
	Set ArrowUpTex = %param7%
	Set ArrowDownTex = %param8%

	########################################

	#Floor indicator
	if[%Direction% = front | %Direction% = Front] AddFloorIndicator %FLInd-Prefix%, front, %CenterX%, %CenterZ% - 0.002, 0.45, 0.333, %Offset%
	if[%Direction% = left | %Direction% = Left] AddFloorIndicator %FLInd-Prefix%, left, %CenterX% - 0.002, %CenterZ%, 0.45, 0.333, %Offset%
	if[%Direction% = right | %Direction% = Right] AddFloorIndicator %FLInd-Prefix%, right, %CenterX% + 0.002, %CenterZ%, 0.45, 0.333, %Offset%
	if[%Direction% = back | %Direction% = Back] AddFloorIndicator %FLInd-Prefix%, back, %CenterX%, %CenterZ% + 0.002, 0.45, 0.333, %Offset%

	#Directional arrows next to floor indicators
	if[%Direction% = front | %Direction% = Front] AddDirectionalIndicator true, true, true, Blank, %ArrowBlankTex%, %ArrowUpTex%, %ArrowBlankTex%, %ArrowDownTex%, %CenterX% + 0.225 - 0.075, %CenterZ% - 0.002, %Offset% + 0.1665 - 0.1875, front, 0.225, 0.375, false, 0, 0
	if[%Direction% = left | %Direction% = Left] AddDirectionalIndicator true, true, true, Blank, %ArrowBlankTex%, %ArrowUpTex%, %ArrowBlankTex%, %ArrowDownTex%, %CenterX% - 0.002, %CenterZ% - 0.225 + 0.075, %Offset% + 0.1665 - 0.1875, left, 0.225, 0.375, false, 0, 0
	if[%Direction% = right | %Direction% = Right] AddDirectionalIndicator true, true, true, Blank, %ArrowBlankTex%, %ArrowUpTex%, %ArrowBlankTex%, %ArrowDownTex%, %CenterX% + 0.002, %CenterZ% + 0.225 - 0.075, %Offset% + 0.1665 - 0.1875, right, 0.225, 0.375, false, 0, 0
	if[%Direction% = back | %Direction% = Back] AddDirectionalIndicator true, true, true, Blank, %ArrowBlankTex%, %ArrowUpTex%, %ArrowBlankTex%, %ArrowDownTex%, %CenterX% - 0.225 + 0.075, %CenterZ% + 0.002, %Offset% + 0.1665 - 0.1875, back, 0.225, 0.375, false, 0, 0
<EndFunction>

	if[%Direction% = front | %Direction% = Front] AddWall Elev(%ElevNo%)IndPanel, %PanelTexture%, 0, %CenterX% - 0.85, %CenterZ% - 0.001, %CenterX% + 0.85, %CenterZ% - 0.001, 0.5, 0.5, %Offset%, %Offset%, 0, 0, false
